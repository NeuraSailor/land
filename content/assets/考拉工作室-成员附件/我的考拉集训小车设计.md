[4-Channel Encoder Motor Drive Module](https://category.yahboom.net/products/quad-md-module)  
根据Yahboom官方的描述，这块4通道编码器电机驱动模块**既支持I2C通信，也支持串口通信（通常通过Type-C口实现UART串口连接）**。你在搭配ESP32使用时，两种方式都可以选择：

- I2C方式：只需要用SDA、SCL接线，与ESP32的I2C接口连接，通过库中对应方法设置相关I2C地址即可。
- Type-C方式（串口UART）：通过Type-C线连接，一般内部转成串口通信（UART），ESP32同样能通过软/硬串口与它通信。

#### 对电机控制  
![[assets/42acaef29915337307d025c1e5afcf25_MD5.jpg|300]]

- 当 $v_1, v_2$ 异号或其中一个为零时: 差动机器人作绕自身中心的自转 (因为力矩中心位于机器人两轮中心)，机器人中处的线速度为零，其角速度为：

$\omega = \frac{2(v_1-v_2)}{H}$

`Adafruit_MotorShield.h` 是 Adafruit 公司开发的一个用于控制直流电机和步进电机的 Arduino 库。这个库主要用于控制 Adafruit Motor Shield v 2 扩展板。

## 主要类和基本函数

### 1. Adafruit_MotorShield 类

这是主要的控制类，用于初始化和管理电机扩展板。

#### 基本构造函数和初始化
```cpp
// 创建电机扩展板对象（默认I2C地址0x60）
Adafruit_MotorShield AFMS = Adafruit_MotorShield();

// 创建带自定义I2C地址的对象
Adafruit_MotorShield AFMS = Adafruit_MotorShield(0x61);

// 初始化扩展板
void begin(uint16_t freq = 1600);  // freq: PWM频率，默认1600Hz
```

#### 获取电机对象
```cpp
// 获取直流电机对象（端口1-4）
Adafruit_DCMotor *getMotor(uint8_t n);

// 获取步进电机对象
Adafruit_StepperMotor *getStepper(uint16_t steps, uint8_t n);
```

### 2. Adafruit_DCMotor 类

用于控制直流电机的类。

#### 基本控制函数
```cpp
// 设置电机速度（0-255）
void setSpeed(uint8_t speed);

// 运行电机
void run(uint8_t direction);
// direction参数：
// FORWARD  - 正转
// BACKWARD - 反转
// RELEASE  - 停止/释放

// 完整的运行控制
void fullOn();   // 全速运行
void fullOff();  // 完全停止
```

### 3. Adafruit_StepperMotor 类

用于控制步进电机的类。

#### 基本控制函数
```cpp
// 设置步进电机转速（RPM）
void setSpeed(uint16_t rpm);

// 单步控制
uint8_t onestep(uint8_t direction, uint8_t style);
// direction: FORWARD 或 BACKWARD
// style: SINGLE, DOUBLE, INTERLEAVE, MICROSTEP

// 多步控制
void step(uint16_t steps, uint8_t direction, uint8_t style);

// 释放电机
void release();
```

## 完整示例代码

### 直流电机控制示例
```cpp
#include <Adafruit_MotorShield.h>

// 创建电机扩展板对象
Adafruit_MotorShield AFMS = Adafruit_MotorShield();

// 获取电机对象（使用M1端口）
Adafruit_DCMotor *myMotor = AFMS.getMotor(1);

void setup() {
  Serial.begin(9600);
  Serial.println("Adafruit Motorshield v2 - DC Motor test!");

  // 初始化扩展板
  if (!AFMS.begin()) {
    Serial.println("Could not find Motor Shield. Check wiring.");
    while (1);
  }
  Serial.println("Motor Shield found.");

  // 设置初始速度
  myMotor->setSpeed(150);
  myMotor->run(FORWARD);
  myMotor->run(RELEASE);
}

void loop() {
  Serial.print("tick");

  // 正转，逐渐加速
  myMotor->run(FORWARD);
  for (int i = 0; i < 255; i++) {
    myMotor->setSpeed(i);
    delay(10);
  }

  // 逐渐减速
  for (int i = 255; i >= 0; i--) {
    myMotor->setSpeed(i);
    delay(10);
  }

  Serial.print("tock");

  // 反转，逐渐加速
  myMotor->run(BACKWARD);
  for (int i = 0; i < 255; i++) {
    myMotor->setSpeed(i);
    delay(10);
  }

  // 逐渐减速并停止
  for (int i = 255; i >= 0; i--) {
    myMotor->setSpeed(i);
    delay(10);
  }

  myMotor->run(RELEASE);
  delay(1000);
}
```

### 步进电机控制示例
```cpp
#include <Adafruit_MotorShield.h>

Adafruit_MotorShield AFMS = Adafruit_MotorShield();

// 连接到Motor Port #1 (M1 and M2) 和 Motor Port #2 (M3 and M4)
// 200步/转的步进电机
Adafruit_StepperMotor *myStepper = AFMS.getStepper(200, 1);

void setup() {
  Serial.begin(9600);
  Serial.println("Stepper test!");

  if (!AFMS.begin()) {
    Serial.println("Could not find Motor Shield. Check wiring.");
    while (1);
  }
  Serial.println("Motor Shield found.");

  myStepper->setSpeed(10);  // 10 RPM
}

void loop() {
  Serial.println("Single coil steps");
  myStepper->step(100, FORWARD, SINGLE);
  myStepper->step(100, BACKWARD, SINGLE);

  Serial.println("Double coil steps");
  myStepper->step(100, FORWARD, DOUBLE);
  myStepper->step(100, BACKWARD, DOUBLE);

  Serial.println("Interleave coil steps");
  myStepper->step(100, FORWARD, INTERLEAVE);
  myStepper->step(100, BACKWARD, INTERLEAVE);

  Serial.println("Microstep steps");
  myStepper->step(50, FORWARD, MICROSTEP);
  myStepper->step(50, BACKWARD, MICROSTEP);

  delay(1000);
}
```

## 主要常量定义

```cpp
// 电机方向
#define FORWARD  1
#define BACKWARD 2
#define BRAKE    3
#define RELEASE  4

// 步进电机模式
#define SINGLE     1
#define DOUBLE     2
#define INTERLEAVE 3
#define MICROSTEP  4
```

这个库使用简单，功能强大，是控制电机的理想选择。记得在使用前先调用 `begin()` 函数初始化扩展板！
